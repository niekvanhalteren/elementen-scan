<script>

const questions = [

  ["aarde","Ik ben met mijn aandacht in het hier en nu."],

  ["aarde","Ik bied stabiliteit wanneer het onrustig wordt."],

  ["aarde","Mensen ervaren mij als betrouwbaar."],

  ["aarde","Ik handel zorgvuldig."],

  ["aarde","Ik ben een rots in de branding."],

 

  ["water","Ik voel goed aan wat er speelt in een groep."],

  ["water","Ik stem mij gemakkelijk af op anderen."],

  ["water","Ik creëer verbinding."],

  ["water","Ik luister zonder direct te willen oplossen."],

  ["water","Ik ben gevoelig voor onderstromen."],

 

  ["lucht","Ik zie patronen en verbanden."],

  ["lucht","Ik kan afstand nemen en reflecteren."],

  ["lucht","Ik breng nieuwe perspectieven in."],

  ["lucht","Ik denk vooruit."],

  ["lucht","Ik werk graag met ideeën."],

 

  ["vuur","Ik neem initiatief."],

  ["vuur","Ik breng energie in beweging."],

  ["vuur","Ik durf besluiten te nemen."],

  ["vuur","Ik geef richting."],

  ["vuur","Ik zet aan tot actie."]

];

 

const form = document.getElementById("form");

form.innerHTML = questions.map((q,i)=>`

  <div class="question">

    <strong>${i+1}. ${q[1]}</strong><br>

    ${[1,2,3,4,5].map(v=>`

      <label>

        <input type="radio" name="q${i}" value="${v}" required> ${v}

      </label>

    `).join("")}

  </div>

`).join("");

 

function calculate(){

  let score={aarde:0,water:0,lucht:0,vuur:0};

  let count={aarde:0,water:0,lucht:0,vuur:0};

 

  questions.forEach((q,i)=>{

    const v=document.querySelector(`input[name="q${i}"]:checked`);

    if(!v){ alert("Beantwoord alle vragen."); return; }

    score[q[0]]+=Number(v.value);

    count[q[0]]++;

  });

 

  let avg={

    aarde:score.aarde/count.aarde,

    water:score.water/count.water,

    lucht:score.lucht/count.lucht,

    vuur:score.vuur/count.vuur

  };

 

  let x=(avg.vuur-avg.water)/4;

  let y=(avg.lucht-avg.aarde)/4;

 

  draw(x,y);

  document.getElementById("result").innerText=

    `Jouw positie: vuur–water = ${x.toFixed(2)}, lucht–aarde = ${y.toFixed(2)}`;

}

 

function draw(x,y){

  const c=document.getElementById("diagram");

  const ctx=c.getContext("2d");

  ctx.clearRect(0,0,500,500);

 

  ctx.strokeStyle="#999";

  ctx.beginPath();

  ctx.moveTo(250,0); ctx.lineTo(250,500);

  ctx.moveTo(0,250); ctx.lineTo(500,250);

  ctx.stroke();

 

  ctx.fillText("Lucht",260,15);

  ctx.fillText("Aarde",260,490);

  ctx.fillText("Water",10,260);

  ctx.fillText("Vuur",460,260);

 

  ctx.beginPath();

  ctx.arc(250+x*200,250-y*200,8,0,Math.PI*2);

  ctx.fill();

}
